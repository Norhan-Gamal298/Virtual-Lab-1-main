##  python Code 

```python
import cv2  # Import OpenCV for image processing
import numpy as np  # Import NumPy for numerical operations
import matplotlib.pyplot as plt  # Import Matplotlib for plotting

# Load image using OpenCV
img = cv2.imread('sample.jpg')  # Load the image (ensure filename is correct)

# Check if the image loaded successfully
if img is None:
    print("Error: Image not found or failed to load.")  # Error message if image not found
else:
    # Resize the image to half its original size
    resized = cv2.resize(img, (img.shape[1] // 2, img.shape[0] // 2))  # Resize for easier handling

    # Convert resized image to RGB for correct display using Matplotlib
    img_rgb = cv2.cvtColor(resized, cv2.COLOR_BGR2RGB)

    # Convert to HSV color space
    img_hsv = cv2.cvtColor(resized, cv2.COLOR_BGR2HSV)
    img_hsv_rgb = cv2.cvtColor(img_hsv, cv2.COLOR_HSV2RGB)  # Convert HSV back to RGB for visualization

    # Convert to YCbCr color space
    img_ycbcr = cv2.cvtColor(resized, cv2.COLOR_BGR2YCrCb)
    img_ycbcr_rgb = cv2.cvtColor(cv2.cvtColor(img_ycbcr, cv2.COLOR_YCrCb2BGR), cv2.COLOR_BGR2RGB)

    # Convert to grayscale
    gray = cv2.cvtColor(resized, cv2.COLOR_BGR2GRAY)

    # Apply Canny edge detection
    edges = cv2.Canny(gray, 100, 200)

    # Plot the original and converted color spaces
    plt.figure(figsize=(18, 6))

    # Display original RGB image
    plt.subplot(1, 3, 1)
    plt.imshow(img_rgb)
    plt.title('Original RGB Image')
    plt.axis('off')

    # Display HSV color image
    plt.subplot(1, 3, 2)
    plt.imshow(img_hsv_rgb)
    plt.title('HSV Color Space')
    plt.axis('off')

    # Display YCbCr color image
    plt.subplot(1, 3, 3)
    plt.imshow(img_ycbcr_rgb)
    plt.title('YCbCr Color Space')
    plt.axis('off')

    plt.tight_layout()
    plt.show()

    # Display edges in a separate figure
    plt.figure(figsize=(6, 5))
    plt.imshow(edges, cmap='gray')
    plt.title('Canny Edge Detection')
    plt.axis('off')
    plt.tight_layout()
    plt.show()


```



##  MATLAB Code

```matlab
% Read the input image
img = imread('sample.jpg');  % Load the image (ensure filename is correct)

% Check if image loaded successfully
if isempty(img)
    disp('Error: Image not found or failed to load.');  % Print error message
else
    % Resize the image to half of its original size
    resized = imresize(img, 0.5);  % Scale image down by 50%

    % MATLAB reads images in RGB by default, so no BGR to RGB conversion needed
    img_rgb = resized;  % Assign resized image to RGB variable

    % Convert RGB to HSV color space
    img_hsv = rgb2hsv(img_rgb);

    % Convert HSV back to RGB for visualization
    img_hsv_rgb = hsv2rgb(img_hsv);

    % Convert RGB to YCbCr color space
    img_ycbcr = rgb2ycbcr(img_rgb);

    % Convert resized image to grayscale for edge detection
    gray = rgb2gray(resized);

    % Apply Canny edge detection (thresholds scaled between 0 and 1)
    edges = edge(gray, 'Canny', [0.1 0.2]);

    % Plot the original RGB image and color space conversions
    figure('Position', [100, 100, 1200, 400]);  % Set figure size

    subplot(1, 3, 1);
    imshow(img_rgb);
    title('Original RGB Image');
    
    subplot(1, 3, 2);
    imshow(img_hsv_rgb);
    title('HSV Color Space');
    
    subplot(1, 3, 3);
    imshow(img_ycbcr);
    title('YCbCr Color Space');

    % Display Canny edges in a separate figure
    figure('Position', [400, 400, 500, 400]);
    imshow(edges);
    title('Canny Edge Detection');
end


```
